/**
     * Método que cria uma Obra.
     * @param evt ActionEvent, Ação do botão sendo clicado
     */
    private void AdicionarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_AdicionarActionPerformed
       Obra aux=null;
        boolean teste = false;
        try {
            if (RLivro1.isSelected()) {
            aux = new Livros(txtNome.getText(), txtAssunto.getText(), txtAutor.getText(), txtEditora.getText(), txtGenero.getText(),Integer.parseInt(txtCopias.getText()),cod);
            teste = true;
        }else if (RPeriodicos1.isSelected()) {
            aux = new Periodicos(txtNome.getText(),txtEditora.getText(),txtAssunto.getText(),txtNumero.getText(),Integer.parseInt(txtCopias.getText()),cod);
            teste = true;
        }else if (RNormas1.isSelected()) {
            aux = new NormasTecnicas(txtNome.getText(), txtAssunto.getText(), txtNumero.getText(),Integer.parseInt(txtCopias.getText()), cod);
            teste = true;
        }else if (RQuadrinhos1.isSelected()) {
            aux = new Quadrinhos(txtNome.getText(), txtAssunto.getText(), txtEditora.getText(), txtAutor.getText(),Integer.parseInt(txtCopias.getText()), cod);
            teste = true;
        }else if (RManga1.isSelected()) {
            aux = new Manga(txtGenero.getText(), txtNome.getText(), txtAssunto.getText(), txtEditora.getText(), txtAutor.getText(),Integer.parseInt(txtCopias.getText()), cod);
            teste = true;
        }else if (RArtigo1.isSelected()) {
            aux = new Artigo(txtNome.getText(), txtAssunto.getText(), txtAutor.getText(), Integer.parseInt(txtCopias.getText()),cod);
            teste = true;
        }else if (RMidias1.isSelected()) {
            aux = new MidiasEletronicas(txtNome.getText(), txtAutor.getText(), txtAssunto.getText(), (String)Tipo.getSelectedItem(),Integer.parseInt(txtCopias.getText()), cod);
            teste = true;
        }else{
            JOptionPane.showMessageDialog(null,"Selecione uma categoria ","Aviso",JOptionPane.ERROR_MESSAGE);
        }
        } catch (java.lang.NumberFormatException e) {
