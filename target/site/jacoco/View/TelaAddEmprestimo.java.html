    public TelaAddEmprestimo(String Admin) { 
        initComponents();
        this.Admin=Admin;
        
        dtmO = (DefaultTableModel)TabelaObra.getModel();    
        arrayO=co.Listar();
        if (!arrayO.isEmpty()) {     
            for (Obra aux: arrayO) {
                String disponivel="Não";
                if (ce.CopiaDisponivel(aux.getNome(),aux.getCopias())) {
                    disponivel="Sim";
                }
            String a[]={aux.getNome(),aux.getAutor(),aux.getAssunto(),aux.getEditora(),aux.getGenero(),aux.getNumero(),aux.getTipo(),disponivel};
            dtmO.addRow(a);
        }
        }
        
        dtmU = (DefaultTableModel)TabelaUsuario.getModel();
        arrayU=cobre.Listar();
        if (!arrayU.isEmpty()) {           
            for (Usuarios aux: arrayU) {
            String a[]={aux.getNome(),aux.getCPF(),aux.getRG(),aux.getTelefone(),aux.getEndereco(),aux.getTipo()};
            dtmU.addRow(a);
        }
        }
    }

/**
     * Método que cria um Emprestimo.
     * @param evt ActionEvent, Ação do botão sendo clicado
     */
    private void AdicionarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_AdicionarActionPerformed
        if (TabelaObra.getSelectedRow()>=0 && TabelaUsuario.getSelectedRow()>=0) {
            String nome = (String) TabelaObra.getValueAt(TabelaObra.getSelectedRow(),0);
            String assunto = (String) TabelaObra.getValueAt(TabelaObra.getSelectedRow(),2);
            String tipo = (String) TabelaObra.getValueAt(TabelaObra.getSelectedRow(),6);
            int copias=0;
            for (Obra o : arrayO) {
                if(o.getNome().equals(nome) && o.getAssunto().equals(assunto) && o.getTipo().equals(tipo)){
                    copias=o.getCopias();
                    break;
                }
            }
            
            if (ce.CopiaDisponivel(nome, copias)) {
                Calendar dtPrevista = Calendar.getInstance();
                String cpf = (String) TabelaUsuario.getValueAt(TabelaUsuario.getSelectedRow(),1);
                for (Usuarios aux : arrayU) {
                if(aux.getCPF().equals(cpf)){
                    if (aux.getDivida()>0) {
                       ce.TelaDivida(this,true,aux.getDivida(), aux);
                    }
                    break;
                }
            } 
                arrayU=cobre.Listar();
            for (Usuarios aux : arrayU) {
                if(aux.getCPF().equals(cpf)){
                    if (aux.getDivida()==0) {
                        dtPrevista.add(Calendar.DATE,aux.getDias());
                        Emprestimo e = new Emprestimo(aux.getNome(), Admin, (String)TabelaObra.getValueAt(TabelaObra.getSelectedRow(),0), dtPrevista, null, Calendar.getInstance());
                        ce.Adcionar(e);
                        dispose();
                    }
                }
            }
            
            }else{
                JOptionPane.showMessageDialog(null,"Cópias indisponíveis","Erro",JOptionPane.WARNING_MESSAGE);
            }
            
        }else{
            JOptionPane.showMessageDialog(null,"Selecione ums Obra e um Usuário","Aviso",JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_AdicionarActionPerformed
